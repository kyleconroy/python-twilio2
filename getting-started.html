Quickstart¶

REST API¶

conn = TwilioClient()
call = conn.calls.make(to="9991231234, from_="9991231234",
                       url="http://foo.com/call.xml")
print call.length
print call.sid

TwiML¶

from twilio import twiml

r = twiml.Response()
r.play("monkey.mp3", loop=5)
r.toxml()
# returns <Response><Play loop="3">monkey.mp3</Play><Response>
